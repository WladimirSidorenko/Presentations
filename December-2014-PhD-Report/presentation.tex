              \tilde{P}(X,Y) = exp\Big\{\sum_i{\Theta_{i} * f_{i}(x = X, y = Y)}\Big\}
            \end{equation}

            \raggedright{\: Normalized probability of the tag sequence:}
            \begin{equation}
              P(Y|X) = \frac{\tilde{P}(X,Y)}{\sum_{Y}\tilde{P}(X,Y)}
            \end{equation}

            \begin{equation}
              \begin{split}
              \Theta & = \argmax_{\Theta}(\prod_{m=1}^M{P(y[m]|x[m])})\\
